aws ssm send-command ^
  --document-name "AWS-RunPowerShellScript" ^
  --comment "Get root (C:) volume free space" ^
  --parameters commands=["Get-Volume -DriveLetter C ^| Select-Object DriveLetter,@{Name='SizeGB';Expression={[math]::Round($_.Size/1GB,2)}},@{Name='FreeGB';Expression={[math]::Round($_.SizeRemaining/1GB,2)}},@{Name='FreePercent';Expression={[math]::Round(($_.SizeRemaining/$_.Size)*100,2)}} ^| ConvertTo-Json -Compress"] ^
  --targets "Key=PlatformTypes,Values=Windows" ^
  --region us-east-1
