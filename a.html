import smtplib from argparse import ArgumentParser from email.message import EmailMessage from email.utils import make_msgid from datetime import datetime from zoneinfo import ZoneInfo from datetime import timedelta def main(): parser = ArgumentParser(description="Send email notification when automated tests fail and await approval.") parser.add_argument('-s', '--sender', required=True, help='From: email header') parser.add_argument('-r', '--recipient', required=True, action='append', default=[], dest='recipients', help='To: email header') parser.add_argument('-c', '--cr', required=True, help='CR number') parser.add_argument('-a', '--application', required=True, help='Application name') parser.add_argument('-pu', '--pipelineurl', required=True, help='GitLab pipeline URL') parser.add_argument('-ps', '--plannedstart', required=True, help='Planned start time (HH:MM)') parser.add_argument('-pd', '--plannedduration', required=True, help='Duration window in minutes') parser.add_argument('-st', '--start', required=True, help='Pipeline start timestamp') args = parser.parse_args() # Time Calculations (matching your template's business logic requirements) startTime = datetime.strptime(args.start, '%Y-%m-%dT%H:%M:%S%fz') startTime = startTime.replace(tzinfo=ZoneInfo('UTC')) eastern_tz = ZoneInfo('America/New_York') now = datetime.now(tz=eastern_tz) time_object_start = datetime.strptime(args.plannedstart, '%H:%M').time() datetime_object_start = now.replace(hour=time_object_start.hour, minute=time_object_start.minute) datetime_object_end = datetime_object_start + timedelta(minutes=int(args.plannedduration)) nowFormatted = now.strftime("%Y-%m-%d %H:%M:%S") plannedStartFormatted = datetime_object_start.strftime("%Y-%m-%d %H:%M:%S") plannedEndFormatted = datetime_object_end.strftime("%Y-%m-%d %H:%M:%S") startEtFormatted = startTime.astimezone(eastern_tz).strftime("%Y-%m-%d %H:%M:%S") msg = EmailMessage() msg['Subject'] = f'⚠️ ACTION REQUIRED BEMSCR-{args.cr} :BEMSD: {args.application} - Automated Tests Failed - PAUSED FOR APPROVAL' msg['To'] = ', '.join(args.recipients) msg['From'] = args.sender msg.preamble = 'Container Rebuild - Test Failure and Hold Alert.\n' # Customized Informational Message Layout postComments = "Automated quality regression testing failed. Deployment sequence halted." holdActionText = "PAUSED AT GATE: The pipeline is waiting for an operator to manually review logs and select Approve/Reject." msg.set_content(f"""\ The automated tests for the following application container have FAILED: Application: {args.application} Status: PAUSED AT APPROVAL GATE - MANUAL INTERVENTION REQUIRED Planned Implementation Window: {plannedStartFormatted} - {plannedEndFormatted} ET Actual Build Execution Start Time: {startEtFormatted} ET Actual Evaluation Check Time: {nowFormatted} ET Current State: Not Implemented (Awaiting Approval Gate Confirmation) Post Implementation Comments: {postComments} Instruction: {holdActionText} Review Action GitLab Pipeline URL: {args.pipelineurl} """) deploy_cid = make_msgid() msg.add_alternative(f"""\

⚠️ Action Required: Deployment Paused

The automated test suite for {args.application} completed with a FAILURE status.

Planned Implementation Window: {plannedStartFormatted} - {plannedEndFormatted} ET

Actual Build Execution Start Time: {startEtFormatted} ET

Actual Evaluation Check Time: {nowFormatted} ET

Implementation Results: Awaiting Action

Comments: {postComments}

👉 NEXT STEPS: Please log in to GitLab, evaluate the test failure logs, and explicitly trigger the manual gate_and_approval_hold button to either force production deployment or cancel execution.

GitLab Pipeline Link: {args.pipelineurl}

""", subtype='html') # Send through your official validated corporate network relay with smtplib.SMTP('mailhost1.cbp.dhs.gov', 25) as s: s.send_message(msg) print("Test failure notification email dispatched successfully via mailhost1.") if __name__ == '__main__': main()
