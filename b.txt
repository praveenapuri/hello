trigger_target_pipeline:
  stage: automationtests-at
  variables:
    RUN_STAGE: $RUN_STAGE
    STAGE_NAME: $STAGE_NAME
    SERENITY_ENV: $SERENITY_ENV
    CUCUMBER_TAGS: $CUCUMBER_TAGS
    CI_COMMIT_TAG: $CI_COMMIT_TAG
  trigger:
    project: 'bems/becn/visualize/becn-dash-webui-at'
    branch: $CI_COMMIT_TAG
    strategy: depend
    forward:
      pipeline_variables: true
  rules:     
    - if: $CI_PIPELINE_SOURCE == "schedule" && $DEV_AUTOMATED == "true"

# APPROVE (only appears when test fails)
approve_deploy:
  stage: approval
  needs: ["trigger_target_pipeline"]
  when: manual
  rules:
    - when: on_failure
  script:
    - echo "Approved to continue deployment"

# REJECT (also appears when test fails)
reject_deploy:
  stage: approval
  needs: ["trigger_target_pipeline"]
  when: manual
  rules:
    - when: on_failure
  script:
    - echo "Deployment rejected"
    - exit 1   # This FAILS the pipeline intentionally
