test_status_collector:
  stage: automationtests-at
  needs:
    - job: trigger_target_pipeline
      artifacts: false
  script:
    - |
      if [ "$CI_PIPELINE_SOURCE" ]; then
        # Check upstream job result via exit status
        if [ "$CI_JOB_STATUS" = "success" ]; then
          echo "TEST_OUTCOME=success" >> result.env
        else
          echo "TEST_OUTCOME=failure" >> result.env
        fi
      fi
  artifacts:
    reports:
      dotenv: result.env
  when: always
