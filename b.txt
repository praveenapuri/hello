# modules/dashboards/variables.tf

variable "aws_region" {
  description = "The AWS region where the resources are located."
  type        = string
  default     = "us-east-1"
}

# --- EC2 Dashboard Variables ---
variable "ec2_dashboards" {
  description = "A map of EC2 dashboard configurations to create. The map key will be used in the dashboard name."
  type = map(object({
    instance_id      = string
    disk_device      = optional(string, "nvme0n1p1")
    disk_path        = optional(string, "/")
    file_system_type = optional(string, "xfs")
  }))
  default = {}
}
