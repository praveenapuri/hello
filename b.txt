###############################################
# Test Status Collector
###############################################
test_status_collector:
  stage: test-evaluation
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  script:
    - echo "Fetching status of the triggered pipeline bridge job..."
    - |
      cat << 'EOF' > parse_status.py
      import sys, json
      try:
          data = json.load(sys.stdin)
          if isinstance(data, list):
              status = next((job['status'] for job in data if isinstance(job, dict) and job.get('name') == 'trigger_target_pipeline'), 'failed')
              print(status)
          else:
              print('failed')
      except Exception:
          print('failed')
      EOF
    - >
      TRIGGER_STATUS=$(curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" 
      "https://gitlab.cbp.dhs.gov/api/v4/projects/$CI_PROJECT_ID/pipelines/$CI_PIPELINE_ID/bridges" 
      | python3 parse_status.py | tr -d '\n\r ')
    - echo "Downstream test pipeline finished with status -> [${TRIGGER_STATUS}]"
    - |
      if [ "$TRIGGER_STATUS" = "success" ]; then
        echo "TEST_OUTCOME=success" > result.env
      else
        echo "TEST_OUTCOME=failure" > result.env
      fi
    - cat result.env
  artifacts:
    reports:
      dotenv: result.env
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: always

# EMAIL NOTIFICATION (ONLY ON FAILURE)
###############################################
approval_notification:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs: 
    - job: test_status_collector
      artifacts: true
  before_script:
    - 'curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" https://gitlab.cbp.dhs.gov/api/v4/projects/19048/repository/files/sendEmail.py/raw?ref=master -o sendEmail.py'
  script:
    - |
      if [ "$TEST_OUTCOME" = "failure" ]; then
        python3 ./sendEmail.py -s "$EMAIL_FROM" -r "$EMAIL_TO" -c "$CR_NUM" -a "$APP_NAME" -ps "$START_TIME" -pd "$DURATION" -st "$CI_PIPELINE_CREATED_AT"
      else
        echo "Tests passed. Skipping email notification."
      fi
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'

###############################################
# APPROVE / REJECT (ONLY USED ON FAILURE)
###############################################
approve_deploy:
  stage: approval
  needs: 
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed automatically. Manual approval not required."
        exit 0
      fi
    - echo "Manual action triggered. Proceeding..."
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: manual
      allow_failure: true  # Setting to true keeps the pipeline green/visible if skipped

reject_deploy:
  stage: approval
  needs:
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed automatically. Reject option skipped."
        exit 0
      fi
    - echo "Deployment explicitly rejected by operator. Halting pipeline."
    - exit 1
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: manual
      allow_failure: true
