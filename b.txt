gate_and_approval_hold:
  stage: approval
  needs:
    - job: test_status_collector
  variables:
    # Define your list of authorized numeric GitLab User IDs separated by spaces
    ALLOWED_USER_IDS: "12345 67890 55443" 
  before_script:
    # Check if the ID of the user who clicked play is in the allowed list
    - |
      if ! echo " $ALLOWED_USER_IDS " | grep -q " $GITLAB_USER_ID "; then
        echo "❌ Access Denied: User ID '$GITLAB_USER_ID' is not authorized to approve this gate."
        exit 1
      fi
  script:
    - echo "✅ Approval cleared by User ID $GITLAB_USER_ID. Releasing deployment sequence..."
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: manual
      allow_failure: false
