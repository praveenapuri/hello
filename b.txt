stages:
  - test
  - approval
  - deploy

# 1. Test job (downstream project)
test_dev:
  stage: test
  trigger:
    project: group/test-project
    branch: test-branch
    strategy: depend

# 2. APPROVE (only appears when test fails)
approve_deploy:
  stage: approval
  needs: ["test_dev"]
  when: manual
  rules:
    - when: on_failure
  script:
    - echo "Approved to continue deployment"

# 3. REJECT (also appears when test fails)
reject_deploy:
  stage: approval
  needs: ["test_dev"]
  when: manual
  rules:
    - when: on_failure
  script:
    - echo "Deployment rejected"
    - exit 1   # ❌ This FAILS the pipeline intentionally

# 4. Auto deploy if tests passed
deploy_stage_auto:
  stage: deploy
  needs: ["test_dev"]
  rules:
    - when: on_success
  script:
    - ./deploy-stage.sh

# 5. Deploy after APPROVAL (if tests failed)
deploy_stage_after_approval:
  stage: deploy
  needs: ["approve_deploy"]
  script:
    - ./deploy-stage.sh
