# Install the installer helper first
Install-Module -Name AWS.Tools.Installer -Force

# Use the helper to install the EC2 module (and the common core module)
Install-AWSToolsModule EC2,Common -Force

# Load the module manually (usually auto-loads, but good for testing)
Import-Module AWS.Tools.EC2

# Check if the command is now recognized
Get-Command Get-EC2Instance

# Set your region
Set-DefaultAWSRegion -Region us-east-1

# Use your CLI profile (replace 'default' with your profile name if different)
Initialize-AWSDefaultConfiguration -ProfileName default
