auto_gate_pass:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed! Marking job as GREEN. Auto-playing manual gate..."
        
        # Fetch the Job ID for gate_and_approval_hold
        GATE_JOB_ID=$(curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "$CI_API_V4_URL/projects/$CI_PROJECT_ID/pipelines/$CI_PIPELINE_ID/jobs" | python3 -c "import sys, json; print(next((j['id'] for j in json.load(sys.stdin) if j.get('name') == 'gate_and_approval_hold'), ''))")
        
        # Auto-Play the manual gate via API
        if [ -n "$GATE_JOB_ID" ]; then
          curl -s -k -X POST -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "$CI_API_V4_URL/projects/$CI_PROJECT_ID/jobs/$GATE_JOB_ID/play"
          echo "Manual gate cleared automatically. Releasing deployments..."
          exit 0
        else
          echo "Gate job not found. Check pipeline structure."
          exit 1
        fi
      else
        echo "Tests failed. Marking job as RED. Manual gate remains locked."
        exit 1
      fi
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      allow_failure: true # Prevents the entire pipeline from immediately crashing, allowing the manual hold to function



approval_notification:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
  before_script:
    - 'curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" https://gitlab.cbp.dhs.gov/api/v4/projects/19048/repository/files/sendEmailTestFailure.py/raw?ref=master -o sendEmailTestFailure.py'
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed. Skipping failure email notification."
        exit 0
      fi
      
      echo "Automated tests failed. Executing dedicated sendEmailTestFailure.py tracking script..."
      python3 ./sendEmailTestFailure.py \
        -s "$EMAIL_FROM" \
        -r "$EMAIL_TO" \
        -c "$CR_NUM" \
        -a "$APP_NAME" \
        -pu "$CI_PIPELINE_URL" \
        -ps "$START_TIME" \
        -pd "$DURATION" \
        -st "$CI_PIPELINE_CREATED_AT"


gate_and_approval_hold:
  stage: approval
  environment:
    name: deployment-approval-gate
    action: prepare
  script:
    - echo "Approval cleared. Releasing deployment sequence..."
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: manual         
      allow_failure: false
