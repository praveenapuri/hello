auto_gate_pass:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed! Automatically triggering the approval gate via API..."
        JOB_ID=$(curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "https://gitlab.cbp.dhs.gov/api/v4/projects/$CI_PROJECT_ID/pipelines/$CI_PIPELINE_ID/jobs?scope=manual" | python3 -c "import sys, json; print(next((j['id'] for j in json.load(sys.stdin) if j['name'] == 'gate_and_approval_hold'), ''))")
        if [ -n "$JOB_ID" ]; then
          curl -X POST -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "https://gitlab.cbp.dhs.gov/api/v4/projects/$CI_PROJECT_ID/jobs/$JOB_ID/play"
          echo "Gate successfully played."
        else
          echo "Manual gate job ID not found."
          exit 1
        fi
      else
        echo "Tests failed. Blocking auto-advance pathway."
        exit 1  # <--- Forces the job to fail visually
      fi
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      allow_failure: true  # <--- Keeps the pipeline alive and turns the job ORANGE
