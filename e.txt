  stage: approval
  needs:
    - job: test_status_collector
    - job: auto_gate_pass 
  environment:
    name: deployment-approval-gate 
    action: prepare
  script:

###############################################
# AUTO-APPROVER (RUNS ON SUCCESS TO BYPASS THE MANUAL GATE)
###############################################
auto_gate_pass:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed! Marking job as GREEN. Auto-playing manual gate..."
        
        # Fetch the Job ID for gate_and_approval_hold
        GATE_JOB_ID=$(curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "$CI_API_V4_URL/projects/$CI_PROJECT_ID/pipelines/$CI_PIPELINE_ID/jobs?per_page=100" | python3 -c "import sys, json; print(next((j['id'] for j in json.load(sys.stdin) if j.get('name') == 'gate_and_approval_hold'), ''))")
        
        # Auto-Play the manual gate via API
        if [ -n "$GATE_JOB_ID" ]; then
          sleep 5
          HTTP_STATUS=$(curl -s -k -o /dev/null -w "%{http_code}" -X POST -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "$CI_API_V4_URL/projects/$CI_PROJECT_ID/jobs/$GATE_JOB_ID/play")
          
          if [ "$HTTP_STATUS" = "200" ] || [ "$HTTP_STATUS" = "201" ]; then
            echo "Success! Manual gate cleared automatically (HTTP $HTTP_STATUS). Releasing deployments..."
            exit 0
          else
            echo "ERROR: Failed to play job. GitLab API returned HTTP status: $HTTP_STATUS."
            exit 1
          fi
        else
          echo "Gate job not found. Check pipeline structure."
          exit 1
        fi
      else
        echo "Tests failed. Marking job as RED. Manual gate remains locked."
        exit 1
      fi
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      allow_failure: true
