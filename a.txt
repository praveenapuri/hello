# 1. The main (publishing) SNS Topic
resource "aws_sns_topic" "topic_a" {
  name = "my-central-publishing-topic"
}

# 2. The secondary (subscribing) SNS Topic
resource "aws_sns_topic" "topic_b" {
  name = "my-specialized-subscribing-topic"
}

# 3. The subscription that links Topic A to Topic B
resource "aws_sns_topic_subscription" "topic_a_to_topic_b" {
  topic_arn = aws_sns_topic.topic_a.arn
  protocol  = "sns"
  endpoint  = aws_sns_topic.topic_b.arn
}

# 4. The resource-based policy attached to Topic B
# This is the critical step that grants Topic A permission to publish to Topic B.
resource "aws_sns_topic_policy" "allow_topic_a_to_publish" {
  arn    = aws_sns_topic.topic_b.arn
  policy = data.aws_iam_policy_document.sns_topic_policy.json
}

# Data source to construct the SNS topic policy document
data "aws_iam_policy_document" "sns_topic_policy" {
  statement {
    effect  = "Allow"
    actions = ["SNS:Publish"]

    principals {
      type        = "Service"
      identifiers = ["sns.amazonaws.com"]
    }

    resources = [
      aws_sns_topic.topic_b.arn,
    ]

    # This condition ensures that only Topic A can publish to Topic B
    condition {
      test     = "ArnEquals"
      variable = "aws:SourceArn"
      values   = [aws_sns_topic.topic_a.arn]
    }
  }
}
