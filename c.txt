###############################################
# Test Status Collector
###############################################
test_status_collector:
  stage: test-evaluation
  needs:
    - job: trigger_automated_test_suite
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  script:
    - echo "Fetching status of the triggered pipeline bridge job..."
    - |
      cat << 'EOF' > parse_status.py
      import sys, json
      try:
          data = json.load(sys.stdin)
          if isinstance(data, list):
              status = next((job['status'] for job in data if isinstance(job, dict) and job.get('name') == 'trigger_automated_test_suite'), 'failed')
              print(status)
          else:
              print('failed')
      except Exception:
          print('failed')
      EOF
    - >
      TRIGGER_STATUS=$(curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" "https://dhs.gov" | python3 parse_status.py | tr -d '\n\r ')
    - echo "Downstream test pipeline finished with status -> [${TRIGGER_STATUS}]"
    - |
      if [ "$TRIGGER_STATUS" = "success" ]; then
        echo "TEST_OUTCOME=success" > result.env
      else
        echo "TEST_OUTCOME=failure" > result.env
      fi
    - cat result.env
  artifacts:
    reports:
      dotenv: result.env
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: always


###############################################
# AUTO-APPROVER (RUNS ON SUCCESS TO BYPASS THE MANUAL GATE)
###############################################
auto_gate_pass:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
  script:
    - |
      if [ "$TEST_OUTCOME" = "success" ]; then
        echo "Tests passed! Marking job as GREEN. Downstream manual gate will be cleanly skipped."
      else
        echo "Tests failed. Marking job as RED to trigger approvals and notifications."
        exit 1
      fi
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      allow_failure: false # <--- Turns red on test failure to wake up downstream stage dependencies



###############################################
# EMAIL NOTIFICATION (PAUSED FOR MANUAL APPROVAL)
###############################################
approval_notification:
  stage: approval
  image: docker.cbp.dhs.gov/cloud/ubi/python:3.11
  needs:
    - job: test_status_collector
      artifacts: true
    - job: auto_gate_pass
  rules:
    - if: '($CI_PIPELINE_SOURCE == "schedule" || $CI_PIPELINE_SOURCE == "web") && $DEV_AUTOMATED == "true"'
      when: on_failure # <--- Executes only when auto_gate_pass exits 1 (fails)
  before_script:
    - 'curl -s -k -H "PRIVATE-TOKEN: $REBUILD_TOKEN" https://gitlab.cbp.dhs.gov/api/v4/projects/19048/repository/files/sendEmailTestFailure.py/raw?ref=master -o sendEmailTestFailure.py'
  script:
    - |
      echo "Automated tests failed. Executing dedicated sendEmailTestFailure.py tracking script..."
      python3 ./sendEmailTestFailure.py \
        -s "$EMAIL_FROM" \
        -r "$EMAIL_TO" \
        -c "$CR_NUM" \
        -a "$APP_NAME" \
        -pu "$CI_PIPELINE_URL" \
        -ps "$START_TIME" \
        -pd "$DURATION" \
        -st "$CI_PIPELINE_CREATED_AT"
